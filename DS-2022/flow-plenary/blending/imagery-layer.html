Update Rendering Rule

function updateRenderingRule(event) {
  let funcName = event.detail;
    
  let rf = new RasterFunction({
    functionName: funcName, // Sea Water Salinity
  });
    
  hycom.renderingRule = rf;
  hycom.title = funcName;
}
Update Mosaic Rule

function updateMosaicRule() {
  let newDepth = depthSlider.values[0];
  document.getElementById("curDepth").innerHTML = newDepth + "m";
    
  let mosaic = hycom.mosaicRule.clone();
  mosaic.multidimensionalDefinition[0].values = newDepth;
    
  currents.mosaicRule = mosaic;
  hycom.mosaicRule = mosaic;
}
